#preparing local image for upload to Docker Hub repository

#1. tagging

#template
docker tag <local-image-name>:<local-tag> <dockerhub-username>/<repository-name>:<version-tag>

docker tag my-dns-server:latest prageeeth/dns-server:v1.0

docker tag my-dhcp-server:latest prageeeth/dhcp-server:v1.0

docker tag my-squid-proxy:latest prageeeth/squid-proxy:v1.0

#2. Docker login
docker login #Log in to Docker Hub (only needed once per session)

#3. push to the docker hub

#template
docker push <dockerhub-username>/<repository-name>:<version-tag>

docker push prageeeth/dns-server:v1.0
docker push prageeeth/dhcp-server:v1.0
docker push prageeeth/squid-proxy:v1.0
